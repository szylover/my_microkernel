// kconfig.h
#define KCONFIG_PMM_BACKEND  0   // 0=bitmap, 1=buddy

// kmain.c
#if KCONFIG_PMM_BACKEND == 0
    pmm_register_backend(pmm_bitmap_get_ops());

    pmm_register_backend(pmm_buddy_get_ops());
#endif
pmm_init();
